SQL Cheat Sheet

Database Queries & Data Management

🔍 Basic Queries

SELECT * FROM users;
Select all columns from table
تحديد جميع الأعمدة من الجدول
SELECT name, email FROM users;
Select specific columns
تحديد أعمدة محددة
SELECT DISTINCT country FROM users;
Select unique values only
تحديد القيم الفريدة فقط
SELECT * FROM users LIMIT 10;
Limit number of results
تحديد عدد النتائج
SELECT COUNT(*) FROM users;
Count total rows
عد إجمالي الصفوف

🎯 Filtering & Conditions

SELECT * FROM users WHERE age > 25;
Filter with condition
التصفية مع شرط
SELECT * FROM users WHERE name LIKE 'John%';
Pattern matching with LIKE
مطابقة الأنماط باستخدام LIKE
SELECT * FROM users WHERE age BETWEEN 18 AND 65;
Range filtering with BETWEEN
تصفية النطاق باستخدام BETWEEN
SELECT * FROM users WHERE country IN ('USA', 'UK');
Multiple value matching
مطابقة قيم متعددة
SELECT * FROM users WHERE email IS NOT NULL;
Check for non-null values
فحص القيم غير الفارغة

📊 Sorting & Grouping

SELECT * FROM users ORDER BY name ASC;
Sort ascending
الترتيب تصاعدياً
SELECT * FROM users ORDER BY age DESC;
Sort descending
الترتيب تنازلياً
SELECT country, COUNT(*) FROM users GROUP BY country;
Group by column
التجميع حسب العمود
SELECT country, COUNT(*) FROM users GROUP BY country HAVING COUNT(*) > 5;
Group with condition
التجميع مع شرط
SELECT * FROM users ORDER BY name, age DESC;
Multiple column sorting
ترتيب أعمدة متعددة

🔗 Joins

SELECT u.name, p.title FROM users u INNER JOIN posts p ON u.id = p.user_id;
Inner join two tables
ربط داخلي لجدولين
SELECT u.name, p.title FROM users u LEFT JOIN posts p ON u.id = p.user_id;
Left join (all users)
ربط أيسر (جميع المستخدمين)
SELECT u.name, p.title FROM users u RIGHT JOIN posts p ON u.id = p.user_id;
Right join (all posts)
ربط أيمن (جميع المنشورات)
SELECT u.name, p.title FROM users u FULL JOIN posts p ON u.id = p.user_id;
Full outer join
ربط خارجي كامل
SELECT u1.name, u2.name FROM users u1, users u2 WHERE u1.manager_id = u2.id;
Self join
ربط ذاتي

➕ Insert Data

INSERT INTO users (name, email) VALUES ('John', 'john@email.com');
Insert single row
إدراج صف واحد
INSERT INTO users (name, email) VALUES ('John', 'john@email.com'), ('Jane', 'jane@email.com');
Insert multiple rows
إدراج صفوف متعددة
INSERT INTO users_backup SELECT * FROM users;
Insert from select
الإدراج من تحديد
INSERT INTO users (name, email) VALUES ('John', 'john@email.com') ON DUPLICATE KEY UPDATE email = VALUES(email);
Insert or update on duplicate
الإدراج أو التحديث عند التكرار
INSERT IGNORE INTO users (name, email) VALUES ('John', 'john@email.com');
Insert ignoring duplicates
الإدراج مع تجاهل المكررات

✏️ Update Data

UPDATE users SET age = 30 WHERE id = 1;
Update specific row
تحديث صف محدد
UPDATE users SET age = age + 1;
Update all rows
تحديث جميع الصفوف
UPDATE users SET age = 25, city = 'NYC' WHERE country = 'USA';
Update multiple columns
تحديث أعمدة متعددة
UPDATE users u JOIN posts p ON u.id = p.user_id SET u.post_count = u.post_count + 1;
Update with join
التحديث مع الربط
UPDATE users SET last_login = NOW() WHERE id = 1;
Update with function
التحديث مع دالة

🗑️ Delete Data

DELETE FROM users WHERE id = 1;
Delete specific row
حذف صف محدد
DELETE FROM users WHERE age < 18;
Delete with condition
الحذف مع شرط
DELETE u FROM users u JOIN inactive_users i ON u.id = i.user_id;
Delete with join
الحذف مع الربط
TRUNCATE TABLE users;
Delete all rows (fast)
حذف جميع الصفوف (سريع)
DELETE FROM users ORDER BY created_at LIMIT 10;
Delete with limit
الحذف مع حد

🏗️ Table Management

CREATE TABLE users (id INT PRIMARY KEY, name VARCHAR(50), email VARCHAR(100));
Create table
إنشاء جدول
ALTER TABLE users ADD COLUMN age INT;
Add column
إضافة عمود
ALTER TABLE users DROP COLUMN age;
Remove column
إزالة عمود
ALTER TABLE users MODIFY COLUMN name VARCHAR(100);
Modify column type
تعديل نوع العمود
DROP TABLE users;
Delete table
حذف الجدول

🔑 Indexes & Constraints

CREATE INDEX idx_email ON users(email);
Create index
إنشاء فهرس
CREATE UNIQUE INDEX idx_email_unique ON users(email);
Create unique index
إنشاء فهرس فريد
ALTER TABLE users ADD CONSTRAINT fk_user_posts FOREIGN KEY (user_id) REFERENCES posts(id);
Add foreign key constraint
إضافة قيد مفتاح خارجي
ALTER TABLE users ADD CONSTRAINT uk_email UNIQUE (email);
Add unique constraint
إضافة قيد فريد
DROP INDEX idx_email ON users;
Drop index
حذف الفهرس

📈 Aggregate Functions

SELECT COUNT(*), AVG(age), MAX(age), MIN(age) FROM users;
Multiple aggregates
تجميعات متعددة
SELECT SUM(salary) FROM employees WHERE department = 'IT';
Sum with condition
المجموع مع شرط
SELECT department, AVG(salary) FROM employees GROUP BY department;
Average by group
المتوسط حسب المجموعة
SELECT COUNT(DISTINCT country) FROM users;
Count unique values
عد القيم الفريدة
SELECT YEAR(created_at), COUNT(*) FROM users GROUP BY YEAR(created_at);
Group by date function
التجميع حسب دالة التاريخ

🔄 Subqueries & Advanced

SELECT * FROM users WHERE age > (SELECT AVG(age) FROM users);
Subquery in WHERE
استعلام فرعي في WHERE
SELECT name, (SELECT COUNT(*) FROM posts WHERE user_id = users.id) as post_count FROM users;
Correlated subquery
استعلام فرعي مترابط
SELECT * FROM users WHERE id IN (SELECT user_id FROM posts WHERE status = 'published');
Subquery with IN
استعلام فرعي مع IN
WITH active_users AS (SELECT * FROM users WHERE last_login > '2023-01-01') SELECT * FROM active_users;
Common table expression (CTE)
تعبير الجدول الشائع (CTE)
SELECT ROW_NUMBER() OVER (ORDER BY age) as row_num, name FROM users;
Window function
دالة النافذة
